## Penggunaan

```ts copy title="Import"
import {
  Navbar,
  NavbarLogo,
  NavbarMenu,
  NavbarMenuItem,
  NavbarMenuDropdown,
  NavbarAction,
} from "@govtechmy/myds-react/navbar";
```

```tsx copy title="Anatomi"
export default () => {
  return (
    <Navbar>
      <NavbarLogo />
      <NavbarMenu>
        <NavbarMenuItem />
        <NavbarMenuDropdown>
          <NavbarMenuItem />
        </NavbarMenuDropdown>
      </NavbarMenu>
      <NavbarActionGroup />
    </Navbar>
  );
};
```
