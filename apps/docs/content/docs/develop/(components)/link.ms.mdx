## Usage

```tsx copy
import { Link } from "@govtechmy/myds-react/link";
```

```tsx copy
export default () => (
  <Link href="https://design.digital.gov.my" primary newTab underline="always">
    {/* Destinasi pautan harus dinyatakan di sini. */}
    Sistem Reka Bentuk Rasmi Malaysia
  </Link>
);
```
