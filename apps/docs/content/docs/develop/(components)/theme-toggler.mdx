## Usage

```tsx copy
import {
  ThemeProvider,
  ThemeToggler,
} from "@govtechmy/myds-react/theme-toggler";
```

```tsx copy
export default () => (
  <ThemeProvider>
    <ThemeToggler/>
  </ThemeProvider>
);
```

## Examples

### Basic Usage

Wrap your application with `ThemeProvider` and use `ThemeToggler` to switch between themes.

### Custom Themes

Pass an array of themes to `ThemeToggler` to customize the available themes.

  ```tsx
 import { MoonIcon, SunIcon } from "@govtechmy/myds-react/icon";
 import {
  ThemeProvider,
  ThemeToggler,
 } from "@govtechmy/myds-react/theme-toggler";

    <ThemeProvider>
      <ThemeToggler
        themes={[
          { theme: "light", icon: <SunIcon /> },
          { theme: "dark", icon: <MoonIcon /> },
          { theme: "sephia", icon: <PlaceholderIcon /> },
          { theme: "retro", icon: <PlaceholderIcon /> },
        ]}
      />
    </ThemeProvider>
  ```
  </Tab>
