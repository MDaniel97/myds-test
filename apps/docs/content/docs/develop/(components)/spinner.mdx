## Usage

```tsx copy
import { Spinner } from "@govtechmy/myds-react/spinner";
```

```tsx copy
export default () => (
    <Spinner>
);
```

## Examples

### Variants

Use the `color` prop to change the spinner style!

<Tabs items={["Preview", "Code"]} defaultValue="Preview">
  <Tab value="Preview">
    <div className="space-y-2">
        <div className="flex items-center justify-center">
            <Button variant="default-outline">Loading Gray <Spinner/></Button>
        </div>
        <div className="flex items-center justify-center">
            <Button>Loading White<Spinner color={"white"}/></Button>
        </div>
        <div className="flex items-center justify-center">
            <Button variant="default-outline">Loading Default <Spinner/></Button>
        </div>
  </div>
  </Tab>
  <Tab value="Code">
  ```tsx 
import { Spinner } from "@govtechmy/myds-react/spinner";

/** Color: Gray **/
<Button variant="default-outline">
    Loading Gray
    <Spinner color={"gray"} />
</Button>

/** Color: White **/
<Button >
    Loading White
    <Spinner color={"white"} />
</Button>

/** Varian: Default **/
<Button variant="default-outline">
    Loading Default
    <Spinner />
</Button>

```
  </Tab>
</Tabs>
