## Usage
```tsx copy
import { Skiplink } from "@govtechmy/myds-react/skiplink";
```

```tsx copy
export default () => (
  <Skiplink href="#main-content">
    <span>Skip to main content</span>
  </Skiplink>
  <main id="main-content">
    <span className="text-slate-600">
      Main content here
    </span>
  </main>
);
```
