## Penggunaan

```ts copy title="Import"
import {
  Footer,
  SiteInfo,
  FooterSection,
  SiteLinkGroup,
  SiteLink,
  FooterLogo,
} from "@govtechmy/myds-react/footer";
```

```tsx copy title="Anatomy"
export default () => (
  <Footer>
    <FooterSection>
      <SiteInfo>
        <FooterLogo />
      </SiteInfo>
      <SiteLinkGroup>
        <SiteLink />
      </SiteLinkGroup>
    </FooterSection>
  </Footer>
);
```
