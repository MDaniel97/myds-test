## Usage

```tsx copy
import { Pill } from "@govtechmy/myds-react/pill";
```

```tsx copy
export default () => <Pill>Trending</Pill>;
```

## Examples

### Size

Use the `size` prop to change the pill size.

<Tabs items={["Preview", "Code"]} defaultValue="Preview">
  <Tab value="Preview" className="not-prose flex justify-center items-end gap-1">
      <Pill size="small">small</Pill>
      <Pill size="medium">medium</Pill>
      <Pill size="large">large</Pill>
  </Tab>
  <Tab value="Code">
    ```tsx 
    import { Pill } from "@govtechmy/myds-react/pill";

    <Pill size="small">small</Pill>;
    <Pill size="medium">medium</Pill>;
    <Pill size="large">large</Pill>;
    ```
  </Tab>
</Tabs>

### With Trailing "x" Button

Pass a callback function to the `onDelete` prop to show a trailing "x" button.

<Tabs items={["Preview", "Code"]} defaultValue="Preview">
  <Tab value="Preview" className="not-prose flex justify-center items-end gap-1">
      <PillWithTrailingXButton>With trailing "x" button</PillWithTrailingXButton>
  </Tab>
  <Tab value="Code">
    ```tsx 
    import { Pill } from "@govtechmy/myds-react/pill";

    <Pill onDelete={...}>With trailing "x" button</Pill>
    ```
  </Tab>
</Tabs>
