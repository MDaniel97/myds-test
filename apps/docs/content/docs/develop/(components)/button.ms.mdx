## Penggunaan

```tsx copy
import {
  Button,
  ButtonIcon,
  ButtonCounter,
} from "@govtechmy/myds-react/button";
```

```tsx copy
export default () => (
  <Button>
    <ButtonIcon />
    <ButtonCounter />
  </Button>
);
```

### Dengan Ikon

Gunakan `<ButtonIcon>` untuk menambah ikon tindakan pada butang.

<Tabs items={["Pratonton", "Kod"]} defaultValue="Pratonton">
  <Tab value="Pratonton">
  <div className="space-y-2">
  <div className="flex justify-center items-center h-fit gap-1">
     <Button variant="default-outline" size="medium">
      <ButtonIcon>
        <ArrowBackIcon />
      </ButtonIcon>
      <span>Go back</span>
    </Button>
     
    <Button variant="default-outline" size="medium">
      <span>Go forward</span>
      <ButtonIcon>
        <ArrowForwardIcon />
      </ButtonIcon>
    </Button>

      </div>

  </div>
  </Tab>
  <Tab value="Kod">
```tsx 
import { Button, ButtonIcon } from "@govtechmy/myds-react/button";
import {
  ArrowForwardIcon,
  ArrowBackIcon,
} from "@govtechmy/myds-react/icon";

/** Dengan Ikon; Posisi: Mula **/

<Button variant="default-outline" size="medium">
  <ButtonIcon>
    <ArrowBackIcon />
  </ButtonIcon>
  <span>Go back</span>
</Button>

/** Dengan Ikon; Posisi: Akhir **/

<Button variant="default-outline" size="medium">
  <span>Go forward</span>
  <ButtonIcon>
    <ArrowForwardIcon />
  </ButtonIcon>
</Button>
```
  </Tab>
</Tabs>
