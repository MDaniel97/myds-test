## Penggunaan

```tsx copy
import { Toggle, ToggleThumb } from "@govtechmy/myds-react/toggle";
```

```tsx copy
export default () => (
  <Toggle>
    <ToggleThumb />
  </Toggle>
);
```

        <Toggle disabled = {true}>

    </div>

        <Toggle disabled = {true} defaultChecked = {true}>
         <ToggleThumb />
        </Toggle>
    </div>

  </div>
  </Tab>

            <ControlledToggle/>
        </div>
    </div>
  </Tab>
