## Usage

```tsx copy
import {
  Table,
  TableBody,
  TableCaption,
  TableCell,
  TableFooter,
  TableHead,
  TableHeader,
  TableRow,
} from "@govtechmy/myds-react/table";
```

```tsx copy
export default () => (
  <Table>
    <TableHeader>
      <TableRow>
        <TableHead />
      </TableRow>
    </TableHeader>
    <TableBody>
      <TableRow>
        <TableCell />
      </TableRow>
    </TableBody>
    <TableCaption />
  </Table>
);
```

## Props

### TableHead

<TypeTable
  type={{
    rowSpan: {
      description: "Number of rows that the cell is to span",
      type: "boolean",
      default: "false",
    },
    colSpan: {
      description: "Number of columns that the cell is to span",
      type: "boolean",
      default: "false",
    },
    scope: {
      description: "Specifies the type of data in the cell",
      type: "col | row | colgroup | rowgroup",
    },
  }}
/>
