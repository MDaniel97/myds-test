## Usage

```tsx copy
import { TextArea } from "@govtechmy/myds-react/textarea";
```

```tsx copy
export default () => (
  <TextArea placeholder="Placeholder text displayed when the text area is empty." 
  size={'small'} 
  disabled={false}
  />
);
```

## Examples
### Size
Controls the input field's size variant. Supports `small`, `medium`, or `large`.
<Tabs items={["Preview", "Code"]} defaultValue="Preview">
  <Tab value="Preview">

      <TextArea size="small" placeholder="Enter your text for your small TextArea."/>
      <TextArea size="medium" placeholder="Enter your text for your medium TextArea."/>
      <TextArea size="large"placeholder="Enter your text for your large TextArea." />
      </div>

  </div>
  </Tab>

/** Size props*/

<TextArea size="small" placeholder="Enter your text for your small TextArea."/>

  </Tab>

### Disabled
The `disabled` props can be used to disable user input and applies a visual styling to indicate the field cannot be edited.
